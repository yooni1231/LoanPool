📁 PDF Google Drive 저장 설정
PDF를 영구적으로 저장하려면 Code.gs에 아래 함수를 추가하세요.
PDF가 Google Drive에 저장되고, 링크로 언제든 열람할 수 있습니다.
Code.gs에 추가할 PDF 저장 코드 보기 ▼ 
// PDF를 Google Drive에 저장하고 URL 반환
function uploadPdfToDrive(base64Data, fileName) {
  try {
    // base64 디코딩 (data:application/pdf;base64, 제거)
    var data = base64Data.replace(/^data:application\/pdf;base64,/, '');
    var blob = Utilities.newBlob(
      Utilities.base64Decode(data),
      'application/pdf',
      fileName
    );
    // "Zivid PDFs" 폴더에 저장 (없으면 자동 생성)
    var folders = DriveApp.getFoldersByName('Zivid PDFs');
    var folder = folders.hasNext()
      ? folders.next()
      : DriveApp.createFolder('Zivid PDFs');
    var file = folder.createFile(blob);
    // 링크 공유 설정 (뷰어 접근 허용)
    file.setSharing(
      DriveApp.Access.ANYONE_WITH_LINK,
      DriveApp.Permission.VIEW
    );
    return { id: file.getId(), url: file.getUrl(), name: fileName };
  } catch(e) {
    return { error: e.message };
  }
}
